include_directories(
  ${CMAKE_CURRENT_SOURCE_DIR}
  )

QT5_WRAP_CPP(moc_files filter_bnpts.h)

set(filter_bnpts_SRCS
  filter_bnpts.cpp
  ${moc_files}
  )

add_library(filter_bnpts SHARED ${filter_bnpts_SRCS})
target_link_libraries(filter_bnpts ${QT_LIBRARIES} ${OPENGL_LIBRARIES} ${CMAKE_THREAD_LIBS_INIT})
install(TARGETS filter_bnpts
  RUNTIME DESTINATION ${BIN_DIR}
  LIBRARY DESTINATION ${LIB_DIR}
  ARCHIVE DESTINATION ${LIB_DIR})
