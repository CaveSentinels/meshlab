include_directories(
  ${CMAKE_CURRENT_SOURCE_DIR}
  ${CMAKE_CURRENT_SOURCE_DIR}/../edit_quality/common
  )

  ../edit_quality/common/transferfunction.cpp

  ../edit_quality/common/meshmethods.cpp
